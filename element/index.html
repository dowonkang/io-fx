<io-fx> element

HTML 예:

<script async defer crossorigin="anonymous" src="https://review.cttd.co.kr/io-fx/io-fx.js"></script>
<io-fx name="fx-name" filter="filter-name" duration="0.5s" delay="0.5s" thresholds="[0.5, 1.0]" easing="ease-in">
  <p>content...</p>
</io-fx>

CSS 예:

/* CSS 변수로 효과 지정 */
io-fx[name="fx-name"] {
  --io-fx-scale-from: 1;
  --io-fx-scale-to: 2;
  --io-fx-opacity-from: 1;
  --io-fx-opacity-to: 0;
}

io-fx[filter="filter-name"] {
  --io-fx-filter-from: blur(0);
  --io-fx-filter-to: blur(10px);
}

/* [active] 셀렉터 사용하여 직접 효과 지정 */
io-fx[name="blink"] .blink {
  opacity: 0;
}

io-fx[name="blink"][active] .blink {
  animation: blink 1s infinite;
}

@keyframes blink {
  0%,
  100% {
    opacity: 0;
  }
  50% {
    opacity: 1;
  }
}
